"""Tests for the kri CLI against a fake kritamcp plugin server.

The fake server records every request body in `FakePlugin.requests_log` and
answers from `FakePlugin.responses` (action -> dict, or list of dicts served
in order for polling tests). Tests invoke the real CLI via subprocess with
KRITA_URL pointed at the fake server."""
import base64
import json
import os
import subprocess
import sys
import tempfile
import threading
import unittest
from http.server import BaseHTTPRequestHandler, HTTPServer

REPO = os.path.dirname(os.path.dirname(os.path.abspath(__file__)))
KRI = os.path.join(REPO, "cli", "kri")


class FakePlugin(BaseHTTPRequestHandler):
    requests_log = []
    responses = {}

    def log_message(self, fmt, *args):
        pass

    def do_GET(self):
        self._reply({"status": "ok", "plugin": "kritamcp"})

    def do_POST(self):
        n = int(self.headers.get("Content-Length", 0))
        cmd = json.loads(self.rfile.read(n))
        FakePlugin.requests_log.append(cmd)
        r = FakePlugin.responses.get(cmd["action"], {"status": "ok"})
        if isinstance(r, list):
            r = r.pop(0) if len(r) > 1 else r[0]
        self._reply(r, 500 if r.get("error") else 200)

    def _reply(self, data, status=200):
        body = json.dumps(data).encode()
        self.send_response(status)
        self.send_header("Content-Type", "application/json")
        self.end_headers()
        self.wfile.write(body)

```python
#!/usr/bin/env python3
"""kri — CLI bridge to the kritamcp plugin running inside Krita.

JSON-over-HTTP to the plugin (default http://localhost:5678). Stdlib only,
so every invocation starts fast and there is no venv.

Env:
  KRITA_URL       plugin endpoint (default http://localhost:5678)
  KRITAMCP_TOKEN  shared token; must match the token set in Krita's env
  KRI_LOOK_PATH   override the default image path used by `kri look`

Exit codes: 0 ok · 1 plugin/connection error · 2 usage error.
"""
import argparse
import base64
import json
import os
import sys
import time
import urllib.error
import urllib.request

KRITA_URL = os.environ.get("KRITA_URL", "http://localhost:5678")
TOKEN = os.environ.get("KRITAMCP_TOKEN", "")

WORKSPACES = ["generation", "upscaling", "live", "animation", "custom"]
JOB_STATES = ["all", "queued", "executing", "finished", "cancelled"]
CONTROL_MODES = [
    "scribble", "line_art", "soft_edge", "canny_edge", "depth", "normal",
    "pose", "segmentation", "blur", "stencil",
    "reference", "style", "composition", "face",
]


def send(action, params=None, timeout=30.0):
    """POST one command. The timeout also travels in the body so the plugin
    stops waiting just before we would (same contract server.py used)."""
    body = json.dumps(
        {"action": action, "params": params or {}, "timeout": timeout}
    ).encode()
    headers = {"Content-Type": "application/json"}
    if TOKEN:
        headers["X-Kritamcp-Token"] = TOKEN
    req = urllib.request.Request(KRITA_URL, data=body, headers=headers)
    try:
        with urllib.request.urlopen(req, timeout=timeout) as r:
            return json.loads(r.read().decode())
    except urllib.error.HTTPError as e:
        try:
            return json.loads(e.read().decode())
        except Exception:
            return {"error": f"HTTP {e.code} from plugin"}
    except urllib.error.URLError as e:
        return {"error": (f"Cannot connect to Krita at {KRITA_URL} — is Krita "
                          f"running with the kritamcp plugin? ({e.reason})")}
    except Exception as e:
        return {"error": str(e)}


def out(result):
    """Print result as JSON and exit. Errors go to stderr with exit code 1 so
    `kri a && kri b` chains stop at the first failure."""
    text = json.dumps(result, indent=1, default=str)
    if isinstance(result, dict) and result.get("error"):
        print(text, file=sys.stderr)
        sys.exit(1)
    print(text)
    sys.exit(0)


def clean(**params):
    """Drop None values so the plugin's own defaults apply."""
    return {k: v for k, v in params.items() if v is not None}


def write_image(payload, path):
    """Decode a plugin canvas payload (data_b64) to a file; return summary."""
    raw = base64.b64decode(payload["data_b64"])
    with open(path, "wb") as f:
        f.write(raw)
    return {"path": path, "width": payload.get("width"),
            "height": payload.get("height"), "format": payload.get("format")}


def look_path(payload, override):
    if override:
        return override
    ext = "png" if payload.get("format") == "png" else "jpg"
    return os.environ.get("KRI_LOOK_PATH") or f"/tmp/kri-canvas.{ext}"


def parse_points(tokens):
    pts = []
    for t in tokens:
        try:
            x, y = t.split(",")
            pts.append([int(x), int(y)])
        except ValueError:
            print(f"kri: invalid point '{t}' (expected X,Y)", file=sys.stderr)
            sys.exit(2)
    return pts


def read_source(file_arg):
    if file_arg == "-":
        return sys.stdin.read()
    with open(file_arg) as f:
        return f.read()


# ----- handlers -----

def cmd_health(a):
    headers = {"X-Kritamcp-Token": TOKEN} if TOKEN else {}
    req = urllib.request.Request(f"{KRITA_URL}/health", headers=headers)
    try:
        with urllib.request.urlopen(req, timeout=5) as r:
            out(json.loads(r.read().decode()))
    except Exception as e:
        out({"error": f"Cannot connect to Krita at {KRITA_URL}: {e}"})


def cmd_undo(a):
    out(send("undo"))


def build_parser():
    p = argparse.ArgumentParser(
        prog="kri",
        description="CLI bridge to Krita (kritamcp plugin). Prints JSON; errors exit 1.",
    )
    sub = p.add_subparsers(dest="cmd", required=True, metavar="command")

    sp = sub.add_parser("health", help="check the plugin is up")
    sp.set_defaults(fn=cmd_health)

    sp = sub.add_parser("undo", help="undo last action")
    sp.set_defaults(fn=cmd_undo)

    return p


def main():
    args = build_parser().parse_args()
    args.fn(args)


if __name__ == "__main__":
    main()
```

    def cmd_exec(self, params):
        """Execute arbitrary Python inside Krita (`kri exec`). DISABLED unless
        Krita was started with KRITAMCP_ALLOW_EXEC=1 in its environment.

        Runs on the main GUI thread like every other action. Consequence: a
        long script blocks Krita's UI until it finishes, and a truly hung
        script CANNOT be preempted — the HTTP client times out and returns an
        error, but Krita keeps executing. The skill tells the model to keep
        scripts short; there is no server-side kill.
        """
        if os.environ.get("KRITAMCP_ALLOW_EXEC", "") != "1":
            return {"error": ("exec is disabled. Start Krita with "
                              "KRITAMCP_ALLOW_EXEC=1 to enable it.")}
        code = params.get("code", "")
        if not code.strip():
            return {"error": "exec: empty code"}
        import io
        import traceback
        from contextlib import redirect_stdout
        doc = self.get_active_document()
        namespace = {
            "Krita": Krita,
            "app": Krita.instance(),
            "doc": doc,
            "view": self.get_active_view(),
            "layer": self.get_active_layer(),
            "QColor": QColor, "QImage": QImage, "QPainter": QPainter,
            "QPen": QPen, "QBrush": QBrush, "QPainterPath": QPainterPath,
            "QPointF": QPointF, "QRectF": QRectF, "Qt": Qt,
        }
        buf = io.StringIO()
        try:
            with redirect_stdout(buf):
                exec(compile(code, "<kri-exec>", "exec"), namespace)
        except Exception:
            return {
                "error": "exec raised an exception",
                "traceback": traceback.format_exc(),
                "stdout": buf.getvalue(),
            }
        self._maybe_refresh(doc)
        result_out = {"status": "ok", "stdout": buf.getvalue()}
        result = namespace.get("result")
        if result is not None:
            try:
                json.dumps(result)
                result_out["result"] = result
            except (TypeError, ValueError):
                result_out["result"] = repr(result)
        return result_out
```

Convention: the script communicates via `print()` (captured stdout) and/or by assigning `result = <json-serializable>`.

def cmd_status(a):
    """Document + full AI overview in ONE round-trip via the plugin's batch."""
    res = send("batch", {"commands": [{"action": "doc_info"},
                                      {"action": "ai_overview"}]}, timeout=30.0)
    if not isinstance(res, dict) or res.get("error"):
        out(res)
    results = (res.get("results") or [{}, {}]) + [{}, {}]
    doc_res, ai_res = results[0] or {}, results[1] or {}
    merged = {}
    merged["doc"] = (doc_res.get("document") if not doc_res.get("error")
                     else {"error": doc_res["error"]})
    if not ai_res.get("error"):
        merged.update({k: v for k, v in ai_res.items() if k != "status"})
    else:
        merged["ai"] = {"error": ai_res["error"]}
    out(merged)
```

def cmd_look(a):
    mode = "full" if a.full else "fast"
    res = send("get_canvas", clean(mode=mode, max_dim=a.max_dim), timeout=120.0)
    if not isinstance(res, dict) or res.get("error"):
        out(res)
    out(write_image(res, look_path(res, a.out)))


def cmd_batch(a):
    try:
        commands = json.loads(read_source(a.file))
    except json.JSONDecodeError as e:
        print(f"kri: invalid batch JSON: {e}", file=sys.stderr)
        sys.exit(2)
    if isinstance(commands, dict):
        commands = [commands]
    res = send("batch",
               clean(commands=commands, review=a.look, stop_on_error=True),
               timeout=120.0)
    if not isinstance(res, dict) or res.get("error"):
        out(res)
    canvas = res.pop("canvas", None)
    if a.look and isinstance(canvas, dict) and canvas.get("data_b64"):
        res["canvas"] = write_image(canvas, look_path(canvas, a.out))
    if res.get("stopped_at") is not None:
        res["error"] = f"batch stopped at command {res['stopped_at']}"
    out(res)

# kri — Krita desde la terminal

`kri` habla con el plugin kritamcp dentro de Krita (`localhost:5678`).
Todo comando imprime JSON. Error → exit 1, así que las cadenas `&&` cortan solas.
`kri --help` y `kri <cmd> --help` listan todo.

## Reglas de oro (minimizar turnos)

1. **Primer comando SIEMPRE `kri status`** — documento + estado AI completo
   (workspace, estilo, `model.architecture`, prompt actual, colas, regiones,
   controles, estilos disponibles) en UNA invocación. No encadenes
   `health` + `ai status` + `ai region list` por separado.

2. **Agrupá todo lo agrupable en `kri batch`** — una invocación = un turno:

   ```bash
   kri batch --look fast <<'EOF'
   [{"action": "set_color", "params": {"color": "#ffffff"}},
    {"action": "fill", "params": {"x": 100, "y": 100, "radius": 80}},
    {"action": "ai_set_prompt", "params": {"positive": "..."}}]
   EOF
   ```

   Las actions son los nombres internos del plugin: `new_canvas`, `set_color`,
   `set_brush`, `stroke`, `fill`, `draw_shape`, `clear`, `undo`, `redo`,
   `ai_set_prompt`, `ai_set_params`, `ai_set_workspace`, `ai_generate`,
   `ai_create_region`, `ai_select_region`, `ai_add_control`.
   El batch PARA en el primer error (exit 1 con `stopped_at`).
   `--look fast` te devuelve el canvas final en el mismo turno.

3. **Para mirar el canvas**: `kri look` escribe la imagen (default
   `/tmp/kri-canvas.jpg`) e imprime el path → leelo con Read. `--full` (PNG a
   resolución completa) SOLO para la revisión final, nunca mientras iterás.

4. **Prompts**: `kri status` te da `ai.model.architecture` (familia del modelo).
   Antes de `kri ai set-prompt` aplicá la skill **krita-ai-prompt-format**
   (natural language vs tags danbooru según familia); nombres propios no
   mega-famosos → **image-prompt-unknown-entities**; antes de
   `kri ai generate` → **image-prompt-sanity-check**.

5. **Generar**: `kri ai generate --wait` bloquea hasta que la cola se vacía.
   NO hagas polling manual con `kri ai jobs`.

6. **`kri exec`** (Python arbitrario dentro de Krita) es la válvula de escape
   para lo que no tiene subcomando. Requiere que Krita haya arrancado con
   `KRITAMCP_ALLOW_EXEC=1`. Namespace disponible: `app`, `doc`, `view`,
   `layer`, `Krita`, `QColor`, `QImage`, `QPainter`, `QPen`, `QBrush`,
   `QPainterPath`, `QPointF`, `QRectF`, `Qt`. `print()` vuelve como `stdout`;
   asigná `result = <algo JSON-serializable>` para devolver datos.
   Scripts CORTOS: corre en el main thread de Krita — bloquea la UI y un
   script colgado NO se puede abortar.

```python
#!/usr/bin/env python3
"""PreToolUse hook (matcher: Bash) for the kri prompt workflow.

Fires on every Bash call. Exits silently (no output at all) unless the
command touches the AI prompt: `kri ai set-prompt` / `kri ai generate`, or
`ai_set_prompt` / `ai_generate` inside a `kri batch` heredoc. When it
matches, it reads the active style's resolved architecture from the plugin
(action ai_status on localhost:5678) and injects a reminder naming the model
family + prompt convention.

Reminder-only: never blocks; any failure degrades to a generic reminder."""
import sys
import os
import json
import urllib.request

NAT = "lenguaje natural (frases descriptivas)"
TAG = "tags danbooru (palabras separadas por comas)"

# Arch.name -> (familia, convención del positivo, negativo)
ARCH = {
    "flux": ("Flux", NAT, "vacío (se ignora)"),
    "flux_k": ("Flux Kontext", NAT, "vacío (se ignora)"),
    "flux2_4b": ("Flux.2", NAT, "vacío (se ignora)"),
    "flux2_9b": ("Flux.2", NAT, "vacío (se ignora)"),
    "chroma": ("Chroma", NAT, "vacío (se ignora)"),
    "zimage": ("Z-Image", NAT, "vacío (Turbo, CFG≈1)"),
    "qwen": ("Qwen-Image", NAT, "opcional, poco peso"),
    "qwen_e": ("Qwen Edit", NAT, "opcional"),
    "qwen_e_p": ("Qwen Edit Plus", NAT, "opcional"),
    "qwen_l": ("Qwen Layered", NAT, "opcional"),
    "sd3": ("SD3 / 3.5", NAT, "soportado, suave"),
    "ernie": ("ERNIE Image", NAT, "opcional"),
    "illu": ("Illustrious", TAG, "worst quality, low quality, ..."),
    "illu_v": ("Illustrious (v-pred)", TAG, "worst quality, low quality, ..."),
    "anima": ("Anima (anime)", TAG, "worst quality, low quality, ..."),
}


def fetch_status():
    token = os.environ.get("KRITAMCP_TOKEN", "")
    url = os.environ.get("KRITA_URL", "http://localhost:5678")
    headers = {"Content-Type": "application/json"}
    if token:
        headers["X-Kritamcp-Token"] = token
    body = json.dumps({"action": "ai_status", "params": {}}).encode()
    req = urllib.request.Request(url, data=body, headers=headers)
    with urllib.request.urlopen(req, timeout=3) as r:
        return json.loads(r.read().decode())


def model_line():
    """Best-effort one-liner about the active model. Empty string on any failure."""
    try:
        st = fetch_status()
    except Exception:
        return ""
    m = st.get("model") or {}
    arch = m.get("architecture")
    ck = m.get("checkpoint")
    info = ARCH.get(arch)
    if info:
        fam, conv, neg = info
        return (f"Modelo activo: architecture='{arch}' ({fam}). "
                f"Convención del positivo: {conv}. Negativo: {neg}. ")
    if arch in ("sdxl", "sd15"):
        return (
            f"Modelo activo: architecture='{arch}' (AMBIGUO: SDXL/SD1.5 corre tanto "
            f"booru —Pony/NoobAI/Animagine— como realista de lenguaje natural). "
            f"Mirá checkpoint='{ck}' y el nombre del estilo para decidir la convención. "
        )
    if arch:
        return f"Modelo activo: architecture='{arch}'. "
    return ""


def main():
    try:
        data = json.loads(sys.stdin.read() or "{}")
    except Exception:
        data = {}
    command = (data.get("tool_input") or {}).get("command", "")

    # Fast path: not a kri prompt command → no output, no HTTP call.
    if "kri" not in command:
        return
    has_set_prompt = "ai set-prompt" in command or "ai_set_prompt" in command
    has_generate = ("ai generate" in command or "ai_generate" in command)
    if not (has_set_prompt or has_generate):
        return

    parts = []
    if has_set_prompt:
        parts.append(
            "Antes de escribir este prompt usá: krita-ai-prompt-format (formatear "
            "según el modelo activo), e image-prompt-unknown-entities (googlear "
            "nombres propios no famosos antes de describirlos)."
        )
    if has_generate:
        parts.append(
            "Antes de generar: corré image-prompt-sanity-check sobre el prompt final "
            "(coherencia con el modelo activo y con lo que pidió el usuario; restos de "
            "ediciones, contradicciones, score tags si la familia los pide, negativo, "
            "nombres propios resueltos)."
        )
    base = " ".join(parts)

    line = model_line()
    msg = (line + base) if line else (
        base + " (No pude leer ai_status; verificá el modelo activo vos.)")

    print(json.dumps({
        "hookSpecificOutput": {
            "hookEventName": "PreToolUse",
            "additionalContext": msg,
        }
    }))


if __name__ == "__main__":
    main()
```

## Quickstart (CLI)

Krita is driven by the `kri` command — no MCP server, no venv.

1. Install the Krita plugin (copy `krita-plugin/kritamcp` into Krita's `pykrita`
   folder and enable it in Settings → Python Plugin Manager, then restart Krita).

3. Check: `kri health` → `{"status": "ok", "plugin": "kritamcp"}`

Everyday flow with Claude Code:

    kri status                 # document + AI Diffusion state, one call
    kri look                   # canvas → /tmp/kri-canvas.jpg
    kri batch <<'EOF'          # many steps, one round-trip
    [{"action": "set_color", "params": {"color": "#ffffff"}},
     {"action": "fill", "params": {"x": 100, "y": 100, "radius": 80}}]
    EOF
    kri ai generate --wait     # generate and block until done

Optional:
- `KRITAMCP_TOKEN` — shared auth token (set for both Krita and your shell).
- `KRITAMCP_ALLOW_EXEC=1` (in Krita's environment) — enables `kri exec`
  (arbitrary Python inside Krita; off by default).
- The Claude Code skill in `claude/skills/kri/` teaches the batch-first workflow.
